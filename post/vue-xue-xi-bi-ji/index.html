v-bind解释

v-bind动态赋值给prop

    <a v-bind:href="url">link</a>
    
    </div>


    <script>
    var vm = new Vue({
        el:'#app',
        data:{
            url:'http://www.baidu.com/'
        }
    })


    
    </script>

    <h1 v-if="awesome">Vue is awesome!</h1>
    <h1 v-else>Oh no 😢</h1>
    
    </div>


    <script>
    var vm = new Vue({
        el:'#app',
        data:{
            awesome:false
        }
    })


    
    </script>
